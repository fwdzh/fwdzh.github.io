## B

如果一个格子有数字，那么它的左边或者它的上面，一定需要至少有一边是全有数字的。

```cpp
void ChatGptDeepSeek()
{
    int n, m;
    cin >> n >> m;
    vector<string> s(n);
    for (int i = 0; i < n; i++)
        cin >> s[i];
    for (int i = 1; i < n; i++)
        for (int j = 1; j < m; j++)
        {
            if (s[i][j] != '0')
            {
                int ok = 0, cnt = 0;
                for (int k = 0; k < i; k++)
                {
                    if (s[k][j] != '0')
                        cnt++;
                }
                ok |= (cnt == i);
                cnt = 0;
                for (int k = 0; k < j; k++)
                {
                    if (s[i][k] != '0')
                        cnt++;
                }
                ok |= (cnt == j);
                if (!ok)
                {
                    cout << "NO\n";
                    return;
                }
            }
        }
    cout << "YES\n";
}
```
